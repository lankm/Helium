UserInterface:=[
    login(): Bool,
    logout():,
    isLoggedIn(): Bool
];

User(loggedIn := false):UserInterface =[
    login(username:String, password: String) := {
        username == "user" && password == "pass"
          ? login := true
          ! login := false
    },
    logout() := {
        loggedIn := false
    },
    isLoggedIn():Bool = {
        isLoggedIn := loggedIn
    }
];

Administrator():User = [
    login(username:String, password: String) := {
        username == "user" && password == "adminw"
          ? login := true
          ! login := false
    },
];
